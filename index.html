  <!-- Include the Pusher JavaScript library -->

  <script src="https://js.pusher.com/7.0/pusher.min.js"></script>

  <script>

    // Enable Pusher logging - don't include this in production

    Pusher.logToConsole = true;


    // Initialize Pusher with your app key and cluster

    const pusher = new Pusher('YOUR_APP_KEY', {

      cluster: 'YOUR_APP_CLUSTER'

    });


    // Subscribe to the channel where events are sent

    const channel = pusher.subscribe('my-channel');


    // Bind to the event triggered by your backend (e.g., new order event)

    channel.bind('new-order', function(data) {

      console.log('Received new order event:', data);


      // Update the image source with the URL received from the event

      const imageUrl = data.imageUrl;

      if (imageUrl) {

        const orderImage = document.getElementById('orderImage');

        orderImage.src = imageUrl;

        orderImage.style.display = 'block';

      }

    });

  </script>
